# 仓库优化建议文档

## 一、仓库结构与代码质量评估

### 1. 整体评估
- **结构清晰度**：仓库结构基本清晰，核心功能脚本、配置文件和自动化工作流分离合理
- **功能完整性**：实现了直播源的收集、转换、合并、更新等核心功能
- **自动化程度**：通过GitHub Actions配置了7个工作流，实现了定时和手动触发的自动化更新
- **文档完善度**：README.md文档详细，包含了功能介绍、使用方法和注意事项
- **最近改进**：
  - 统一了M3U和TXT解析函数的4K频道判断逻辑，确保一致性
  - 改进了CCTV 4K/8K频道名称规范化逻辑，支持"CCTV 4K超高清"等变体格式转换为标准"CCTV4K"
  - 修复了分组标题错误分类为4K频道的问题，仅根据频道名称判断4K属性
  - 改进了4K频道判断逻辑，排除含否定词的频道（如"不包含4K"等）

- **代码重复**：多个脚本中存在相似的功能实现（如直播源下载、解析逻辑）
- **错误处理**：部分网络请求和文件操作缺乏完善的异常处理
- **性能瓶颈**：缺少并发处理和缓存机制，处理大量直播源时效率较低
- **模块化不足**：部分功能耦合度高，不利于维护和扩展
- **配置管理**：虽然有统一的sources.json，但部分脚本仍直接使用硬编码的数据源

#### 1.1 建立统一的核心模块
**问题**：多个脚本中存在重复的功能实现，如网络请求、M3U解析、文件处理等

**建议**：
- 创建`core`目录，提取公共功能模块
- 实现以下核心模块：
  - `core/network.py`：统一的网络请求处理，包含重试机制和错误处理
  - `core/parser.py`：M3U和TXT格式的解析与转换
  - `core/file_utils.py`：文件读写、格式转换等通用功能
  - `core/channel_utils.py`：频道分类、标准化、质量筛选等功能

**示例实现**：
```python
# core/network.py
import requests
import time
from typing import Optional, Dict

DEFAULT_HEADERS = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
}

def fetch_content(url: str, retries: int = 3, timeout: int = 10, headers: Optional[Dict] = None) -> Optional[str]:
    """统一的网络请求函数，包含重试机制"""
    if headers is None:
        headers = DEFAULT_HEADERS
    
    for attempt in range(retries):
        try:
            response = requests.get(url, timeout=timeout, headers=headers)
            response.raise_for_status()
            response.encoding = 'utf-8'
            return response.text
        except requests.RequestException as e:
            print(f"请求失败 {url} (尝试 {attempt+1}/{retries}): {e}")
            if attempt < retries - 1:
                time.sleep(2 ** attempt)  # 指数退避
    return None
```

# core/logging_config.py
import logging
import os
from datetime import datetime

def setup_logging(log_file: str = None, log_level: str = 'INFO') -> logging.Logger:
    """配置日志系统"""
    if log_file is None:
        log_file = f'iptv_update_{datetime.now().strftime("%Y%m%d")}.log'
    
    # 创建日志目录
    log_dir = 'logs'
    os.makedirs(log_dir, exist_ok=True)
    
    logger = logging.getLogger('iptv')
    logger.setLevel(getattr(logging, log_level.upper()))
    
    # 避免重复添加处理器
    if not logger.handlers:
        # 文件处理器
        file_handler = logging.FileHandler(os.path.join(log_dir, log_file), encoding='utf-8')
        file_handler.setFormatter(logging.Formatter('%(asctime)s - %(name)s - %(levelname)s - %(message)s'))
        
        # 控制台处理器
        console_handler = logging.StreamHandler()
        console_handler.setFormatter(logging.Formatter('%(levelname)s - %(message)s'))
        
        logger.addHandler(file_handler)
        logger.addHandler(console_handler)
    
    return logger
```

# core/quality_check.py
import requests
import time
from typing import Dict, Tuple

def check_stream_quality(url: str, timeout: int = 5) -> Tuple[bool, Dict]:
    """检查直播流质量"""
    result = {
        'available': False,
        'response_time': None,
        'status_code': None,
        'error': None
    }
    
    try:
        start_time = time.time()
        response = requests.head(url, timeout=timeout, allow_redirects=True)
        result['response_time'] = time.time() - start_time
        result['status_code'] = response.status_code
        
        # 检查是否为有效的流媒体
        if response.status_code == 200:
            content_type = response.headers.get('Content-Type', '')
            if any(media_type in content_type for media_type in ['video/', 'audio/', 'application/vnd.apple.mpegurl']):
                result['available'] = True
                
    except requests.RequestException as e:
        result['error'] = str(e)
    
    return result
```

#### 2.3 实现直播源去重机制
**问题**：当前缺少有效的直播源去重机制，可能导致大量重复内容

**建议**：
- 基于频道名称和URL的双重去重
- 实现智能合并相似直播源的功能
- 支持用户自定义去重规则

**示例实现**：
```python
# core/deduplication.py
from typing import List, Dict, Set
import hashlib

def deduplicate_channels(channels: List[Dict]) -> List[Dict]:
    """直播源去重"""
    seen = {
        'name_url': set(),  # 频道名称+URL的组合
        'url_only': set()   # 仅URL，用于完全重复的直播源
    }
    
    unique_channels = []
    
    for channel in channels:
        name = channel.get('name', '')
        url = channel.get('url', '')
        
        # 计算唯一标识符
        name_url_key = f"{name}|{url}"
        url_key = url
        
        # 检查是否已存在
        if name_url_key not in seen['name_url'] and url_key not in seen['url_only']:
            seen['name_url'].add(name_url_key)
            seen['url_only'].add(url_key)
            unique_channels.append(channel)
    
    return unique_channels
```

### 3. 性能优化

#### 3.1 实现并发处理
**问题**：当前脚本主要使用单线程处理，下载和解析大量直播源时效率较低

**建议**：
- 使用多线程或异步IO处理网络请求
- 实现任务队列和线程池管理
- 优化内存使用，避免一次性加载过多数据

**示例实现**：
```python
# core/concurrent.py
import asyncio
import aiohttp
from typing import List, Dict

async def fetch_url_async(url: str, session: aiohttp.ClientSession, timeout: int = 10) -> str:
    """异步下载URL内容"""
    try:
        async with session.get(url, timeout=timeout) as response:
            response.raise_for_status()
            return await response.text()
    except Exception as e:
        print(f"异步请求失败 {url}: {e}")
        return None

async def fetch_multiple_urls_async(urls: List[str]) -> Dict[str, str]:
    """并发下载多个URL"""
    async with aiohttp.ClientSession() as session:
        tasks = []
        for url in urls:
            task = asyncio.create_task(fetch_url_async(url, session))
            tasks.append((url, task))
        
        results = {}
        for url, task in tasks:
            result = await task
            if result:
                results[url] = result
        
        return results
```

#### 3.2 实现缓存机制
**问题**：每次运行脚本都需要重新下载所有直播源，浪费带宽和时间

**建议**：
- 实现本地缓存，避免重复下载未变化的直播源
- 使用ETag或Last-Modified头检查文件是否更新
- 支持缓存过期时间配置

**示例实现**：
```python
# core/cache.py
import os
import json
import hashlib
from datetime import datetime, timedelta

class CacheManager:
    """缓存管理器"""
    
    def __init__(self, cache_dir: str = 'cache', cache_expiry: int = 24):
        self.cache_dir = cache_dir
        self.cache_expiry = cache_expiry  # 小时
        os.makedirs(self.cache_dir, exist_ok=True)
    
    def get_cache_key(self, url: str) -> str:
        """生成缓存键"""
        return hashlib.md5(url.encode()).hexdigest()
    
    def get_cache_file(self, key: str) -> str:
        """获取缓存文件路径"""
        return os.path.join(self.cache_dir, f"{key}.json")
    
    def is_cached(self, url: str) -> bool:
        """检查是否已缓存"""
        cache_file = self.get_cache_file(self.get_cache_key(url))
        if not os.path.exists(cache_file):
            return False
        
        # 检查缓存是否过期
        with open(cache_file, 'r', encoding='utf-8') as f:
            cache_data = json.load(f)
        
        cached_time = datetime.fromisoformat(cache_data['cached_time'])
        return (datetime.now() - cached_time) < timedelta(hours=self.cache_expiry)
    
    def get_cache(self, url: str) -> str:
        """获取缓存内容"""
        cache_file = self.get_cache_file(self.get_cache_key(url))
        if self.is_cached(url):
            with open(cache_file, 'r', encoding='utf-8') as f:
                cache_data = json.load(f)
            return cache_data['content']
        return None
    
    def set_cache(self, url: str, content: str) -> None:
        """设置缓存"""
        cache_key = self.get_cache_key(url)
        cache_file = self.get_cache_file(cache_key)
        
        cache_data = {
            'url': url,
            'content': content,
            'cached_time': datetime.now().isoformat()
        }
        
        with open(cache_file, 'w', encoding='utf-8') as f:
            json.dump(cache_data, f, ensure_ascii=False, indent=2)
```
